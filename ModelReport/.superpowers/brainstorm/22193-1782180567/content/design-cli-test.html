CLI 设计

python -m model_report \
  --model model.pkl # 评分卡文件路径 (必需)
  --data score_result.csv # 打分结果 (必需)
  --output report.xlsx # 输出路径 (默认: ./model_report.xlsx)
  --metadata vars.yaml # 变量清单 (可选)
  --config custom.toml # 自定义配置 (可选，覆盖默认列名)

Python API

from model_report import ReportGenerator, ReportConfig, PickledScorecardAdapter

config = ReportConfig() # 全部默认
scorecard = PickledScorecardAdapter("model.pkl")
generator = ReportGenerator(scorecard, config)
generator.generate(df).to_excel("output.xlsx")
